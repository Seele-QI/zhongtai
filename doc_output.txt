### 核心功能需求表
| 功能模块 | 功能描述 | 优先级 | 类型 | 备注 |
|---------|--------|------|-----|------|

| 深度链接生成 | 生成包含视频信息的分享URL：domain/share?vid=xxx&title=xxx | P0 | 基础 | 支持URL参数加密 |
| 中转重定向 | 用户扫描二维码→HTTP 302重定向到抖音创作者中心 | P0 | 基础 | 同时保存参数到Redis |
| 参数解析存储 | 解析URL参数，存储到数据库/缓存供前端调用 | P0 | 基础 | 支持过期时间设置 |
| 视频库管理 | 管理预制视频文件，支持上传、编辑、删除、查询 | P1 | 业务 | 关联元数据（标题、文案、封面） |
| 账号绑定 | 管理抖音账号Cookie、登录状态、关联用户 | P1 | 业务 | Cookie加密存储 |
| 自动填充执行 | Playwright/油猴脚本读取参数自动填充抖音表单 | P1 | 业务 | 支持两套方案 |
| 发布日志追踪 | 记录每次QR扫描、参数解析、发布操作的日志 | P2 | 运维 | 用于数据分析 |
| 多账号支持 | 支持同一用户管理多个抖音账号 | P2 | 扩展 | 账号隔离存储 |
| 错误恢复机制 | 自动重试、异常回滚、状态同步 | P2 | 扩展 | 配置最大重试次数 |

# app/services/qr_code_service.py
from typing import Dict, Optional
from dataclasses import dataclass
import qrcode
import base64
from io import BytesIO
@dataclass
class QRCodeRequest:
video_id: str
title: str
description: str
tags: Optional[list] = None
class QRCodeService:
"""QR码生成服务"""
def __init__(self, config, cache_repo, encryption_svc):
self.config = config
self.cache_repo = cache_repo
self.encryption_svc = encryption_svc
def generate(self, request: QRCodeRequest) -> Dict[str, str]:
"""
生成QR码及分享URL
流程:
1. 构建分享URL: https://domain/share/{share_token}
2. 加密参数存入Redis (TTL=24h)
3. 生成QR码图片 (base64)
返回:
{
"share_token": "abc123xyz",
"share_url": "https://domain/share/abc123xyz",
"qr_code_base64": "data:image/png;base64,iVBORw0KGgo..."
}
"""
# Step 1: 生成唯一令牌
share_token = self._generate_share_token(request.video_id)
# Step 2: 加密参数
payload = {
"video_id": request.video_id,
"title": request.title,
"description": request.description,
"tags": request.tags or []
}
encrypted_payload = self.encryption_svc.encrypt(payload)
# Step 3: 存入Redis (TTL=24小时)
ttl_seconds = 24 * 3600
self.cache_repo.set(
key=f"share:{share_token}",
value=encrypted_payload,
ttl=ttl_seconds
)
# Step 4: 构建分享URL
share_url = f"{self.config.APP_URL}/api/v1/share/{share_token}"
# Step 5: 生成QR码
qr_code_base64 = self._encode_qr_code(share_url)
return {
"share_token": share_token,
"share_url": share_url,
"qr_code_base64": qr_code_base64
}
def _generate_share_token(self, video_id: str) -> str:
"""生成唯一的分享令牌"""
import uuid
import time
return f"{video_id}_{int(time.time())}_{uuid.uuid4().hex[:8]}"
def _encode_qr_code(self, data: str) -> str:
"""将数据编码为QR码 (base64)"""
qr = qrcode.QRCode(
version=1,
error_correction=qrcode.constants.ERROR_CORRECT_L,
box_size=10,
border=2,
)
qr.add_data(data)
qr.make(fit=True)
img = qr.make_image(fill_color="black", back_color="white")
# 转换为base64
buffer = BytesIO()
img.save(buffer, format='PNG')
buffer.seek(0)
img_base64 = base64.b64encode(buffer.read()).decode()
return f"data:image/png;base64,{img_base64}"

# app/services/share_service.py
class ShareService:
"""分享链接解析和重定向服务"""
def __init__(self, config, cache_repo, encryption_svc, log_repo):
self.config = config
self.cache_repo = cache_repo
self.encryption_svc = encryption_svc
self.log_repo = log_repo
def resolve_share(self, share_token: str, user_agent: str, ip_addr: str) -> Dict:
"""
解析分享链接，获取视频参数
流程:
1. 从Redis读取加密的参数
2. 检查过期时间
3. 解密参数
4. 记录分享日志
5. 返回解密后的视频参数
返回:
{
"video_id": "video_001",
"title": "视频标题",
"description": "视频文案",
"tags": ["标签1", "标签2"]
}
"""
# Step 1: 从Redis读取数据
encrypted_payload = self.cache_repo.get(f"share:{share_token}")
if not encrypted_payload:
raise ShareExpiredException(f"Share token expired: {share_token}")
# Step 2: 解密
payload = self.encryption_svc.decrypt(encrypted_payload)
# Step 3: 记录日志
log_entry = {
"share_token": share_token,
"video_id": payload["video_id"],
"user_agent": user_agent,
"ip_address": ip_addr,
"status": "scanned"
}
self.log_repo.create(log_entry)
return payload
def get_redirect_url(self, share_token: str) -> str:
"""
获取重定向URL (302 → Douyin Creator)
返回: https://creator.douyin.com/creator-micro/content/upload
"""
return self.config.DOUYIN_UPLOAD_URL

# app/services/playwright_service.py
from playwright.async_api import async_playwright
class PlaywrightService:
"""Playwright自动化填充服务"""
def __init__(self, config, account_repo):
self.config = config
self.account_repo = account_repo
async def auto_fill_and_publish(self, account_id: str, video_info: dict):
"""
使用Playwright自动填充和发布视频
参数:
- account_id: 抖音账号ID
- video_info: {video_id, title, description, tags}
流程:
1. 获取账号的加密Cookie
2. 启动浏览器
3. 加载Cookie登录
4. 导航到创作者中心
5. 上传视频
6. 自动填充标题、描述、标签
7. 返回发布状态
"""
account = self.account_repo.get_by_id(account_id)
if not account:
raise AccountNotFound(f"Account {account_id} not found")
# 解密Cookie
cookies = self._decrypt_cookies(account.cookie_encrypted)
async with async_playwright() as p:
browser = await p.chromium.launch(headless=True)
context = await browser.new_context()
# 加载Cookie
await context.add_cookies(cookies)
page = await context.new_page()
try:
# 导航到创作者中心
await page.goto(self.config.DOUYIN_UPLOAD_URL)
# 检查登录状态
if "login" in page.url or "passport" in page.url:
raise NotLoggedInException("Cookie expired")
# 上传视频文件
await self._upload_video(page, video_info["video_id"])
# 等待上传完成
await page.wait_for_timeout(5000)
# 填充标题
await self._fill_title(page, video_info["title"])
# 填充描述
await self._fill_description(page, video_info["description"])
# 填充标签
if video_info.get("tags"):
await self._fill_tags(page, video_info["tags"])
return {
"status": "ready",
"message": "请在浏览器中点击【发布】按钮"
}
finally:
await browser.close()

await file_input.set_input_files(f"/videos/{video_id}.mp4")
async def _fill_title(self, page, title: str):
"""填充视频标题"""
selectors = [
'input[placeholder*="标题"]',
'input[placeholder*="请输入"]'
]
